var i: Int = 0;
var j: Int = 1;

if (i == j)
{
    print("First condition met\n");
} else
{
    print("First condition not met\n");
}

if (i != j)
{
    print("Second condition met\n");
} else
{
    print("Second condition not met\n");
}

var k: Int = 10;

while (i < k)
{
    print("Number should increment from 0 to 9: ", i, "\n");
    i = i + 1;
}

for (var l: Int = 0; l < 10; l = l + 1)
{
    print("Number should increment from 0 to 9: ", l, "\n");
}

var arr: Int[16][32][64];
arr[10][18][60] = 10;
arr[12][20][61] = 20;
print("Should be 10: ", arr[10][18][60], "\n",
    "Should be 20: ", arr[12][20][61], "\n",
    "Value is not set: ", arr[14][22][61], "\n");

arr[10][18][60] = 30;
arr[12][20][61] = 40;
print("Should be 30: ", arr[10][18][60], "\n",
    "Should be 40: ", arr[12][20][61], "\n",
    "Value is not set: ", arr[14][24][61], "\n");

var printMap: Int[10][20];
printMap[2 + 2][10 + 5] = 1;

for (var y: Int = 0; y < 20; y = y + 1)
{
    for (var x: Int = 0; x < 10; x = x + 1)
    {
        print(printMap[x][y]);
    }
    print("\n");
}
