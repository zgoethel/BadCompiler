var i: Int = 50;
print("Before ", i, "\n");

fun helloWorld(): Nothing
{
    var j: Int = i;
    i = i * 2;
    print("Inside ", i, ", ", j, "\n");
}

print("After declaration ", i, "\n");
helloWorld();
print("After invoke ", i, "\n");
helloWorld();
print("After invoke ", i, "\n");
helloWorld();
print("After invoke ", i, "\n");
